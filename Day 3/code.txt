
def solveTwo(input_):
    parsedInput = input_
    while len(parsedInput) != 1:
        for pos in range(len(list(zip(*parsedInput)))):
            counter = Counter(list(zip(*parsedInput))[pos])
            mostCommon = counter.most_common(1)
            if len(mostCommon) == 1:
                mostCommonDigit = mostCommon[0][0]
            else:
                mostCommonDigit = "1"
            parsedInput = list(
                filter(lambda d: d[pos] == mostCommonDigit, parsedInput))
    oxygenValue = parsedInput[0]
    parsedInput = input_
    while len(parsedInput) != 1:
        for pos in range(len(list(zip(*parsedInput)))):
            counter = Counter(list(zip(*parsedInput))[pos])
            leastCommon = counter.most_common()[:-1-1:-1]
            if len(leastCommon) == 1:
                leastCommonDigit = leastCommon[0][0]
            else:
                leastCommonDigit = "1"
            parsedInput = list(
                filter(lambda d: d[pos] == leastCommonDigit, parsedInput))
    carbonDioxideValue = parsedInput[0]
    return int(carbonDioxideValue, base=2) * int(oxygenValue, base=2)
